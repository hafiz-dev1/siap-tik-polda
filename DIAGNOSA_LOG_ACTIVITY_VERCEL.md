```typescript
// Run this script to check production DB
import { prisma } from './src/lib/prisma';

async function debugProductionDB() {
  console.log('üîç Checking Production Database...\n');

  try {
    // 1. Check connection
    await prisma.$connect();
    console.log('‚úÖ Database connected\n');

    // 2. Check if table exists
    const tables = await prisma.$queryRaw`
      SELECT table_name 
      FROM information_schema.tables 
      WHERE table_schema = 'public' 
        AND table_name = 'activity_log';
    `;
    console.log('üìã Tables found:', tables);

    // 3. Check columns
    const columns = await prisma.$queryRaw`
      SELECT column_name, data_type 
      FROM information_schema.columns 
      WHERE table_name = 'activity_log'
      ORDER BY ordinal_position;
    `;
    console.log('\nüìä Columns in activity_log:');
    console.table(columns);

    // 4. Check if can read
    const count = await prisma.activityLog.count();
    console.log(`\nüìà Total logs: ${count}`);

    // 5. Get sample logs
    const sampleLogs = await prisma.activityLog.findMany({
      take: 5,
      orderBy: { createdAt: 'desc' },
      include: {
        user: {
          select: { username: true, nama: true },
        },
      },
    });

    console.log('\nüìù Sample logs:');
    sampleLogs.forEach((log, i) => {
      console.log(`\n${i + 1}. [${log.category}/${log.type}]`);
      console.log(`   User: ${log.user?.username}`);
      console.log(`   Description: ${log.description}`);
      console.log(`   Time: ${log.createdAt}`);
    });

    console.log('\n‚úÖ All checks passed!');
  } catch (error) {
    console.error('\n‚ùå Error:', error);
    
    if (error instanceof Error) {
      console.error('\nüìã Error details:');
      console.error('Message:', error.message);
      console.error('Stack:', error.stack);
    }
  } finally {
    await prisma.$disconnect();
  }
}

debugProductionDB();
```
